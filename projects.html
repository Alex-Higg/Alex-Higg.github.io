SELECT 
  users.name,
  COUNT(orders.id) as order_count,
  SUM(orders.total) as total_spent
FROM users
JOIN orders ON users.id = orders.user_id
WHERE orders.status = 'completed'
GROUP BY users.id
HAVING COUNT(orders.id) > 5
ORDER BY total_spent DESC
LIMIT 10;
